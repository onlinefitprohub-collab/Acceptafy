import type { HistoryItem, GradingResult } from '../types';
const HISTORY_KEY = 'inboxAuthorityHistory';
const MAX_HISTORY_ITEMS = 20;
export const getHistory = (): HistoryItem[] => { try { const h = localStorage.getItem(HISTORY_KEY); return h ? JSON.parse(h) : []; } catch { return []; } };
export const saveAnalysis = (content: any, result: GradingResult): HistoryItem[] => { 
    const item: HistoryItem = { id: Date.now().toString(), date: new Date().toISOString(), content, result };
    let h = getHistory(); h.unshift(item); if(h.length > MAX_HISTORY_ITEMS) h = h.slice(0, MAX_HISTORY_ITEMS);
    localStorage.setItem(HISTORY_KEY, JSON.stringify(h)); return h; 
};
export const deleteHistoryItem = (id: string): HistoryItem[] => { const h = getHistory().filter(i => i.id !== id); localStorage.setItem(HISTORY_KEY, JSON.stringify(h)); return h; };
export const clearHistory = (): HistoryItem[] => { localStorage.removeItem(HISTORY_KEY); return []; };

Tsx
import React, { useState, useEffect } from 'react';
import { EmailInput } from './components/EmailInput';
import { ResultsHub } from './components/ResultsHub';
import { ResultsTabs } from './components/ResultsTabs';
import { ReputationDashboard } from './components/ReputationDashboard';
import { AcademyHub } from './components/Academy/AcademyHub';
import { HistoryPanel } from './components/HistoryPanel';
import { gradeCopy, generateDnsRecords } from './services/geminiService';
import { getHistory, saveAnalysis } from './services/historyService';
import { AcademyIcon, Logo } from './components/icons/CategoryIcons'; // Assuming Logo is exported here or create a placeholder

const App = () => {
    const [viewState, setViewState] = useState('input');
    const [body, setBody] = useState("Hi [Name],\n\nCheck out our sale!");
    const [variations, setVariations] = useState([{ subject: "Sale!", previewText: "Look inside" }]);
    const [result, setResult] = useState<any>(null);
    const [history, setHistory] = useState<any[]>([]);
    const [showAcademy, setShowAcademy] = useState(false);
    const [dnsRecords, setDnsRecords] = useState<any>(null);
    const [loading, setLoading] = useState(false);

    useEffect(() => { setHistory(getHistory()); }, []);

    const handleGrade = async () => {
        setLoading(true);
        try {
            const res = await gradeCopy(body, variations);
            setResult(res);
            setHistory(saveAnalysis({ body, variations }, res));
            setViewState('results');
        } catch (e: any) { alert(e.message); }
        setLoading(false);
    };

    const handleDns = async (domain: string) => {
        setLoading(true);
        try { setDnsRecords(await generateDnsRecords(domain)); } catch (e) { console.error(e); }
        setLoading(false);
    };

    return (
        <div className="min-h-screen dark-bg text-white font-sans">
            <header className="p-6 border-b border-white/10 flex justify-between items-center backdrop-blur-md sticky top-0 z-10">
                <h1 className="text-2xl font-bold bg-clip-text text-transparent bg-gradient-to-r from-purple-400 to-pink-400">Inbox Authority</h1>
                <button onClick={() => setShowAcademy(true)} className="flex gap-2 items-center px-4 py-2 bg-white/10 rounded-lg hover:bg-white/20"><AcademyIcon /> Academy</button>
            </header>
            <main className="container mx-auto p-6 max-w-5xl">
                <EmailInput body={body} setBody={setBody} variations={variations} setVariations={setVariations} onGrade={handleGrade} isLoading={loading} />
                {viewState === 'results' && result && (
                    <div className="space-y-8 mt-8 animate-fade-in">
                        <ResultsHub scoreData={result.inboxPlacementScore} gradeData={result.overallGrade} />
                        <ResultsTabs result={result} body={body} />
                        <ReputationDashboard history={history} onGenerateDns={handleDns} isGeneratingDns={loading} dnsRecords={dnsRecords} />
                    </div>
                )}
                {viewState === 'results' && <HistoryPanel history={history} />}
            </main>
            {showAcademy && <AcademyHub onClose={() => setShowAcademy(false)} />}
        </div>
    );
};
export default App;
